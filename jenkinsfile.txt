pipeline {
    agent any
    tools {
        maven 'mvn'
    }
    stages {
        stage('Fetching') {
            steps {
                git branch: 'main', url: "https://github.com/PRATAPMUKHERJEE/SpringPetClinic.git"
            }
        }
        stage('Compile') {
            steps {
                sh 'mvn compile'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Package') {
            steps {
                sh 'mvn package'
            }
        }
        stage('Verify') {
            steps {
                sh 'mvn verify'
            }
        }
        stage('Install') {
            steps {
                sh 'mvn install'
            }
        }
    }
}

pipeline {
    agent any
    tools {
        maven 'mvn'
    }
    stages {
        stage('Fetching') {
            steps {
                git branch: 'main', url: "https://github.com/PRATAPMUKHERJEE/SpringPetClinic.git"
            }
        }
 stage('validate')
        {
            steps 
            {
                echo 'Validating the Maven Project'
                sh 'mvn validate'
            }
        }
        stage('clean')
        {
            steps 
            { echo 'Cleaning the Maven Project'
                sh 'mvn clean'
            }
        }
        stage('Compile') {
            steps {
                sh 'mvn compile'
            }
        }
        stage('Test') {
            steps {
                sh 'mvn test'
            }
        }
        stage('Package') {
            steps {
                sh 'mvn package'
            }
        }
        stage('Verify') {
            steps {
                sh 'mvn verify'
            }
        }
        stage('Install') {
            steps {
                sh 'mvn install'
                 }
                  }
stage('executing generate jar'){
steps{
echo 'Executing the generate jar file'
sh 'java -jar ./target/Sum.jar 10 25 30 55 67 89'
            }
        }
    }
}
